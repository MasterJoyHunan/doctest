import { ref, onMounted } from 'vue'
import BScroll from '@better-scroll/core'

const bs = ref({})
const scrollRef = ref()

onMounted(() => {
    bs.value = new BScroll(scrollRef.value, {
        disableMouse: true,
        disableTouch: true
    })
})

let index = 0
let inc = 0
const showLength = 7
const maxLength = showLength * 2 + 2

const list = ref(new Array(maxLength))

const add = () => {
    if (index == maxLength) {
        bs.value.scrollTo(0, 0, 0)
        index = showLength + 1
    }
    list.value[index] = `Vue3.0 无缝滚动组件展示数据第${inc}条`
    if (index > showLength) {
        bs.value.scrollBy(0, -50, 200)
        list.value[index - showLength - 1] = `Vue3.0 无缝滚动组件展示数据第${inc}条`
    }
    index++
    inc++
}
</script>
